## 1. –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞: —ç–∫—Ä–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏–µ –¥–ª—è MarkdownV2

## 2. –≠—Ñ—Ñ–µ–∫—Ç ¬´–ø–µ—á–∞—Ç–∞—é—â–µ–≥–æ—Å—è —Ç–µ–∫—Å—Ç–∞¬ª —Å –ø–æ–¥–¥–µ—Ä–∂–∫–æ–π —Ä–∞–∑–º–µ—Ç–∫–∏

from aiogram.types import Message

async def animate_typing(
    bot: Bot,
    chat_id: int,
    full_text: str,
    parse_mode: str = "MarkdownV2",
    delay: float = 0.1,
    start_with_message: Message = None
):
    """
    –ê–Ω–∏–º–∏—Ä—É–µ—Ç –ø–æ—è–≤–ª–µ–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ —Å –ø–æ–¥–¥–µ—Ä–∂–∫–æ–π MarkdownV2.
    full_text ‚Äî —É–∂–µ —ç–∫—Ä–∞–Ω–∏—Ä–æ–≤–∞–Ω–Ω–∞—è –∏ —Ä–∞–∑–º–µ—á–µ–Ω–Ω–∞—è —Å—Ç—Ä–æ–∫–∞.
    –ï—Å–ª–∏ –ø–µ—Ä–µ–¥–∞–Ω start_with_message, —Ä–µ–¥–∞–∫—Ç–∏—Ä—É–µ—Ç –µ–≥–æ, –∏–Ω–∞—á–µ –æ—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –Ω–æ–≤–æ–µ.
    """

    # –£–ø—Ä–æ—â—ë–Ω–Ω–æ: —Ä–∞–∑–±–∏–≤–∞–µ–º –ø–æ –ø—Ä–æ–±–µ–ª–∞–º, —Å–æ—Ö—Ä–∞–Ω—è—è –∏—Ö.
    parts = full_text.split(' ')

    for i, part in enumerate(parts):
        # –î–æ–±–∞–≤–ª—è–µ–º –ø—Ä–æ–±–µ–ª –º–µ–∂–¥—É —á–∞—Å—Ç—è–º–∏, –∫—Ä–æ–º–µ –ø–µ—Ä–≤–æ–π
        if i > 0:

        await asyncio.sleep(delay)
    return start_with_message
```

## 3. –≠—Ñ—Ñ–µ–∫—Ç ¬´–º–∏–≥–∞–Ω–∏—è¬ª —Å –ø–æ–º–æ—â—å—é —Å–ø–æ–π–ª–µ—Ä–∞

def tokenize_markdownv2(text: str):
    """–†–∞–∑–¥–µ–ª—è–µ—Ç —Ç–µ–∫—Å—Ç –Ω–∞ –∫—É—Å–∫–∏, –Ω–µ —Ä–∞–∑—Ä—ã–≤–∞—è Markdown-–∫–æ–Ω—Å—Ç—Ä—É–∫—Ü–∏–∏."""
    tokens = []
    i = 0
    while i < len(text):
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞—á–∞–ª–æ —É–ø—Ä–∞–≤–ª—è—é—â–µ–π –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
        if text[i] == '*' and i+1 < len(text) and text[i+1] != ' ':
            # –ò—â–µ–º –∑–∞–∫—Ä—ã–≤–∞—é—â—É—é *
            end = text.find('*', i+1)
            if end != -1:
                tokens.append(text[i:end+1])
                i = end + 1
                continue
        elif text[i] == '_' and i+1 < len(text) and text[i+1] != ' ':
            end = text.find('_', i+1)
            if end != -1:
                tokens.append(text[i:end+1])
                i = end + 1
                continue
        elif text[i:i+2] == '||':
            end = text.find('||', i+2)
            if end != -1:
                tokens.append(text[i:end+2])
                i = end + 2
                continue
        # –û–±—ã—á–Ω—ã–π —Å–∏–º–≤–æ–ª
        tokens.append(text[i])
        i += 1
    return tokens
```
